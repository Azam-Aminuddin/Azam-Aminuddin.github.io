class Model_CRMT:
    def __init__(self, params ):
        # Running Variable
        self.Prev_Field_Liquid = 0
        self.Prev_Month = 0
        
        # Variables which subject of the Optimization
        self.params = params
        
    def calculate_field_liquid( self, series ):
        theta, alpha, beta = self.params
        
        ans = series["Field Inj"] + ( self.Prev_Field_Liquid-series["Field Inj"])*NP.exp( -( series["Month"] - self.Prev_Month )/ theta )
        self.Prev_Field_Liquid = ans
        self.Prev_Month = series["Month"]
        return ans

    def calculate_field_oil( self, series ):
        theta, alpha, beta = self.params
        
        ans = series["Estimated-Field Liquid"]/(1 + alpha * NP.power( 10., -10 ) * NP.power( series["Cum Inj"], beta ))
        return ans

## The Objective Function. This is subject to be minimized
def fObjective( params ):
    model = Model_CRMT( params )

    df["Estimated-Field Liquid"] = df.apply( model.calculate_field_liquid, axis = 1 )
    df["Estimated-Error Liquid"] = NP.abs( df["Estimated-Field Liquid"] - df["Field Liquid"] ) * 100. / df["Field Liquid"]

    df["Estimated-Field Oil"] = df.apply( model.calculate_field_oil, axis = 1 )
    df["Estimated-Error Oil"] = NP.abs( df["Estimated-Field Oil"] - df["Field Oil"] ) / df["Field Oil"]
    
    sum = NP.sum( df["Estimated-Error Liquid"]**2) + NP.sum( df["Estimated-Error Oil"]**2)
    return NP.sqrt( sum )

grad_fun = autograd.grad(fObjective)
hessian_fun = autograd.hessian(fObjective)
tol=1e-09

params = NP.array( [1.4372, 152.77, 1.5581] )
print("Initial objective:", fObjective(params))
for i in range(100):
    delta = autograd.numpy.linalg.solve( hessian_fun(params), -grad_fun(params))
    params = params + delta
    if autograd.numpy.linalg.norm(delta) < tol:
        break

print("Trained objective:", fObjective(params))
print("params:", params )
